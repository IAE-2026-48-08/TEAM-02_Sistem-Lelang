Respon sukses harus memuat `status`, `message`, `data`, dan `meta` (berisi `service_name` dan `api_version`). Sementara respon gagal memuat `status`, `message`, dan `errors`.

Cara paling rapi dan profesional di Laravel untuk menangani ini adalah dengan membuat sebuah **Trait**. Trait ini nantinya bisa dipakai di semua *Controller* kamu, sehingga kamu tidak perlu mengetik ulang struktur JSON yang panjang di setiap *endpoint*.

Berikut adalah langkah pembuatannya:

### 1. Buat Trait `ApiResponser`
Buat folder baru bernama `Traits` di dalam folder `app/`. Lalu, buat file baru dengan nama `ApiResponser.php` (`app/Traits/ApiResponser.php`) dan isi dengan kode berikut:

```php
<?php

namespace App\Traits;

use Illuminate\Http\JsonResponse;

trait ApiResponser
{
    /**
     * Membangun format respon sukses (2xx) sesuai Standard Integration Contract
     */
    protected function successResponse($data, $message = 'Data retrieved successfully', $code = 200, $meta = []): JsonResponse
    {
        $defaultMeta = [
            'service_name' => 'Penawaran-Service', // Identitas service kamu
            'api_version'  => 'v1',
        ];

        return response()->json([
            'status'  => 'success',
            'message' => $message,
            'data'    => $data,
            'meta'    => array_merge($defaultMeta, $meta),
        ], $code);
    }

    /**
     * Membangun format respon error (4xx/5xx) sesuai Standard Integration Contract
     */
    protected function errorResponse($message, $code, $errors = null): JsonResponse
    {
        return response()->json([
            'status'  => 'error',
            'message' => $message,
            'errors'  => $errors,
        ], $code);
    }
}
```

### 2. Update `app/Http/Controllers/BidController.php`
Sekarang ubah anotasi di atas setiap *method* menggunakan gaya *Attribute*. Format ini tidak lagi membutuhkan penutup komentar bintang-bintang, melainkan ditulis seperti struktur fungsi bawaan PHP.

```php
<?php

namespace App\Http\Controllers;

use App\Models\Bid;
use Illuminate\Http\Request;
use App\Traits\ApiResponser;
use OpenApi\Attributes as OA;

class BidController extends Controller
{
    use ApiResponser;

    // 1. Endpoint: Collection (GET /api/v1/bids)
    #[OA\Get(
        path: "/api/v1/bids",
        summary: "Mengambil daftar semua penawaran lelang",
        security: [["ApiKeyAuth" => []]],
        responses: [
            new OA\Response(response: 200, description: "Berhasil mengambil riwayat penawaran")
        ]
    )]
    public function index()
    {
        return $this->successResponse(Bid::all(), 'Daftar penawaran berhasil diambil.');
    }

    // 2. Endpoint: Resource (GET /api/v1/bids/{id})
    #[OA\Get(
        path: "/api/v1/bids/{id}",
        summary: "Mengambil detail penawaran spesifik berdasarkan ID",
        security: [["ApiKeyAuth" => []]],
        parameters: [
            new OA\Parameter(
                name: "id",
                in: "path",
                required: true,
                description: "ID dari data penawaran",
                schema: new OA\Schema(type: "integer")
            )
        ],
        responses: [
            new OA\Response(response: 200, description: "Detail penawaran ditemukan"),
            new OA\Response(response: 404, description: "Data penawaran tidak ditemukan")
        ]
    )]
    public function show($id)
    {
        $bid = Bid::find($id);
        if (!$bid) return $this->errorResponse('Penawaran tidak ditemukan', 404);
        return $this->successResponse($bid, 'Detail penawaran.');
    }

    // 3. Endpoint: Action (POST /api/v1/bids)
    #[OA\Post(
        path: "/api/v1/bids",
        summary: "Mengajukan penawaran lelang baru",
        security: [["ApiKeyAuth" => []]],
        requestBody: new OA\RequestBody(
            required: true,
            content: new OA\JsonContent(
                properties: [
                    new OA\Property(property: "item_id", type: "string", example: "ITEM-991"),
                    new OA\Property(property: "bid_amount", type: "number", example: 5500000)
                ]
            )
        ),
        responses: [
            new OA\Response(response: 201, description: "Penawaran baru berhasil diajukan"),
            new OA\Response(response: 400, description: "Validasi data gagal atau tidak lengkap")
        ]
    )]
    public function store(Request $request)
    {
        $validated = $request->validate([
            'item_id' => 'required|string',
            'bid_amount' => 'required|numeric'
        ]);

        $bid = Bid::create([
            'item_id' => $validated['item_id'],
            'user_id' => 'USR-112',
            'bid_amount' => $validated['bid_amount'],
            'status' => 'winning'
        ]);

        return $this->successResponse($bid, 'Penawaran berhasil diajukan.', 201);
    }
}
```
